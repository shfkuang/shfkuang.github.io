\section{变分自编码器推导}
\begin{align*}
\log p_\theta (x^{(i)})
&=\log p_\theta (x^{(i)}) \int_z q_{\phi}  (z | x^{(i)}) dz\\
&=  \int_z q_{\phi}  (z | x^{(i)}) \log \frac{p_\theta (x^{(i)}) p_{\theta}  (z | x^{(i)}) q_{\phi}  (z | x^{(i)})}{q_{\phi}  (z | x^{(i)}) p_{\theta}  (z | x^{(i)})} dz\\
&=  \int_z q_{\phi}  (z | x^{(i)}) \log \frac{p_\theta (x^{(i)}) p_{\theta}  (z | x^{(i)}) q_{\phi}  (z | x^{(i)})}{q_{\phi}  (z | x^{(i)}) p_{\theta}  (z | x^{(i)})} dz\\
&=  \int_z q_{\phi}  (z | x^{(i)}) \log \frac{p_\theta (z,x^{(i)}) q_{\phi}(z | x^{(i)}) }{ q_{\phi}  (z | x^{(i)}) p_{\theta}  (z | x^{(i)})} dz\\
&= \int_z q_{\phi}(z | x^{(i)}) \log \frac{p_\theta (z,x^{(i)})}{ q_{\phi}(z | x^{(i)})} dz + \int_z q_{\phi}(z | x^{(i)}) \log \frac{q_\phi (z | x^{(i)})}{ p_{\theta}(z | x^{(i)})} dz\\
&= L(\theta, \phi; x^{(i)}) + D_{KL}(q_{\phi}(z | x^{(i)}) || p_{\theta}  (z | x^{(i)}))
\end{align*}

\begin{align*}
L(\theta, \phi; x^{(i)})
&= \int_z q_{\phi}(z | x^{(i)}) \log \frac{p_\theta (z,x^{(i)})}{ q_{\phi}(z | x^{(i)})} dz\\
&= \int_z q_{\phi}(z | x^{(i)}) \log \frac{p_\theta (z) p_\theta (x^{(i)} | z)}{ q_{\phi}(z | x^{(i)})} dz\\
&= \int_z q_{\phi}(z | x^{(i)}) \log \frac{p_\theta (z) p_\theta (x^{(i)} | z)}{ q_{\phi}(z | x^{(i)})} dz \\
&= \int_z q_{\phi}(z | x^{(i)}) \log \frac{p_\theta (z) }{ q_{\phi}(z | x^{(i)})} dz +  \int_z q_{\phi}(z | x^{(i)}) \log p_\theta (x^{(i)} | z) dz \\
&= -D_{KL}(q_{\phi}(z | x^{(i)}) || p_\theta (z)) + E_{q_{\phi}(z | x^{(i)})}[\log p_\theta (x^{(i)} | z) ]
\end{align*}

由于KL距离为非负，故
\begin{align*}
p_\theta (x^{(i)}) 
&\geq L(\theta, \phi; x^{(i)})\\
&= E_{q_{\phi}(z | x^{(i)})} [ -\log q_{\phi}(z|x^{(i)}) + \log  p_{\theta} (x^{(i)}, z) ]\\
&= -D_{KL}(q_{\phi}(z | x^{(i)}) || p_\theta (z)) + E_{q_{\phi}(z | x^{(i)})}[\log p_\theta (x^{(i)} | z) ]
\end{align*}
